  <script>
    const API_URL = "https://survivor-ligamx-bot.onrender.com";

    async function loadPicks() {
      try {
        const res = await fetch(`${API_URL}/picks/latest`, {cache: "no-store"});
        const data = await res.json();
        
        document.getElementById("bot-status").textContent = data.status === "active" ? "🟢 Activo" : "🔴 Error/Inactivo";
        document.getElementById("last-sync").textContent = data.last_update ? new Date(data.last_update).toLocaleString() : "-";
        
        const tbody = document.getElementById("picks-tbody");
        tbody.innerHTML = "";
        
        if (!data.picks || data.picks.length === 0) {
          tbody.innerHTML = `<tr><td colspan="5" style="text-align:center;">Sin picks válidos (VIG<15% + EV>4%)</td></tr>`;
          return;
        }

        data.picks.forEach(p => {
          const evClass = (p.expected_value > 0.04) ? "ev-pos" : "ev-neg";
          tbody.innerHTML += `
            <tr>
              <td>${p.match || "N/A"}</td>
              <td>${(p.true_prob * 100).toFixed(1)}%</td>
              <td class="${evClass}">${(p.expected_value * 100).toFixed(2)}%</td>
              <td>${p.kelly_stake?.toFixed(1) || 0}%</td>
              <td>${p.market || "1X2"}</td>
            </tr>`;
        });
      } catch (e) {
        console.error("❌ Fallo API:", e);
        document.getElementById("bot-status").textContent = "🟡 Offline";
        document.getElementById("picks-tbody").innerHTML = `<tr><td colspan="5" style="text-align:center;color:#ff5555;">Error conectando a Render. Revisa internet o espera cold-start (30s).</td></tr>`;
      }
    }

    setInterval(loadPicks, 21600000); // 6 horas
    loadPicks(); // Carga inicial
  </script>
